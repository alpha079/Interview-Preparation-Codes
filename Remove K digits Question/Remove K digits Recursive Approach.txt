#include<bits/stdc++.h>
using namespace std;
void count_smallest_number(string str, int k, string &res)
{
    if(k==0)
     { res.append(str);return;}

    int len=str.length();
    if(len<=k)return;

    int idx=0;
    for(int i=1;i<=k;i++)
    {
        if(str[i]<str[idx])
          {
              idx=i;
          }
    }
    res+=str[idx];
count_smallest_number(str.substr(idx+1,len-idx),k-idx,res);

}

string remove_k_digit(string str,int k)

{
    if(k==0)return str;

    if(str.length()==k)
      return "";

    string res="";
    count_smallest_number(str,k, res);
    return res;

}
